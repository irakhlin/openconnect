    cmake_minimum_required(VERSION 3.8)

find_package(LibXml2)
find_package(ZLIB)
find_package(GnuTLS)
find_package(PkgConfig)

        "${P11_INCLUDE_DIRS}"
        "${PSKC_INCLUDE_DIRS}")

        "${DEPENDENCY_INCLUDES}")
